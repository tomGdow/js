class Color {
  constructor(r, g, b) {
    this.values = [r, g, b]
  }

  getRed() {
    return this.values[0];
  }

  setRed(x) {
    this.values[0] = x;
  }
}

let red = new Color(255, 0, 0);

console.log(JSON.stringify(red.values))

red.setRed(128)
console.log(JSON.stringify(red.values))

Code Example (static)

static
The static keyword is used within classes to define methods or properties that belong to the class itself, rather than to instances of the class
[JavaScript Inserts] 
class Color {

  #values;

  static isValid(r, g, b) {
    return r >= 0 && r <= 255 && 
      g >= 0 && g <= 255 && 
      b >= 0 && b <= 255; 
    } 
  
  constructor(r, g, b) { this.#values=[r, g, b] }
    get getRed() {
      return this.#values[0];
    } 
    set setRed(x) { 
      if (x < 0 || x> 255) {
        throw new RangeError("invalid R Value")
      }
      this.#values[0] = x;
    }

}

let red = new Color(255, 0, 0);

Code Example (extends)

super
The super keyword is used inside classes to access and call functions on an object's parent class
There are two main ways super is used:
  1. Call the parent constructor
  2. Call methods from the parent class
[JavaScript Inserts] 
class Color {

  #values;

  static isValid(r, g, b) {
    return r >= 0 && r <= 255 && 
           g >= 0 && g <= 255 && 
           b >= 0 && b <= 255;
  } 
    
    constructor(r, g, b) { this.#values=[r, g, b] }
            
      get getRed() { 
        return this.#values[0];
      } 
            
    set setRed(x) { 
    
      if (x < 0 || x > 255) {
        throw new RangeError("invalid R Value")
      }

      this.#values[0] = x;
    }

}

class ColorWithAlpha extends Color {
  #alpha;

  constructor(r, g, b, a) {
 
    super(r, g, b);
            
    this.#alpha = a;
  }
            
  get alpha () {
    return this.#alpha;
  }

  set alpha(value) {
    if (value < 0 || value> 1) {
      throw new RangeError(
        "Alpha value must be between 0 and 1"
      )
    }
                
    this.#alpha = value;
  }

}

let red = new Color(255, 0, 0);
red.setRed = 255;
console.log(red.getRed)

let redAlpha = new ColorWithAlpha(255, 0, 0, 0.5)

class Color {

  #values;

  constructor(r, g, b) {
    this.#values = [r, g, b]
  }

  get getRed() {
    return this.#values[0];
  }

  set setRed(x) {

    if (x < 0 || x > 255) {
      throw new RangeError("invalid R Value")
    }
    
    this.#values[0] = x;
  }

  toString() {
    return this.#values.join(", ")
  }
    
  stringify() {
    return JSON.stringify(this.#values)
  }
}

Code Example (extends)

[JavaScript Inserts] 
class MyFirstName {

  #fname

  constructor(fname) {
  this.#fname = fname;
  }

  get firstname() {
    return this.#fname;
  }

  set firstname(x) {
  this.#fname = x;
  }

  getStringSeparator() {
    return "--"
  }

}

class MyFullName extends MyFirstName {

  #lname;

  constructor(fname, lname) {
    super(fname)
    this.#lname = lname
  }

  get lastname() {
    return this.#lname;
  }

  set lastname(x) {
    this.#lname = x;
  }

  fullName() {
    return `${this.firstname}
            ${super.getStringSeparator()}
            ${this.#lname}`
  }

}

let mfn = new MyFullName("Joe", "Bloggs")

console.log(mfn.fullName())
console.log(mfn)
console.log(mfn.firstname)
console.log(mfn.lastname)

mfn.firstname = "Jolly";
mfn.lastname= "Bloggy"

console.log(mfn.fullName())

console.log(mfn instanceof MyFullName)
console.log(mfn instanceof MyFirstName)
