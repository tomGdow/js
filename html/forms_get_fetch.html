    async function sendData() {

      const formData = new FormData(form_one);

      let param1 = formData.get('param1')
      let param2 = formData.get('param2')
      let url = addQueryStringToUrl(
        "https://phptwo.tomgdow.com", 
        "param1", param1,
        "param2", param2);

      try {
        const response = await fetch(url, {
          method: "GET",
        });

        let result = await response.text();
        let responseArray = result.split('&');

        console.log("response", responseArray)

        document.querySelector("#submitted-url").value = url
        document.querySelector("#out-name").value = 
          responseArray[0];
        document.querySelector("#out-email").value = 
          responseArray[1];
        document.querySelector('#returned-querystring').value = 
          responseArray[3] + "&" + responseArray[4];
        document.querySelector('#server-response').value = 
          responseArray[0] + "&" + responseArray[1] + 
            "&" + responseArray[2] + "&";
        document.querySelector('#server-response-two').value = 
          responseArray[3] + "&" + responseArray[4];

      } catch (e) {
        console.error(e);
      }
    }

    document.getElementById('form_one').addEventListener('submit', 
      function (event) {

        event.preventDefault();
        sendData();

        let msg = "Greetings from PHP Server (GET Method)"
        document.getElementById('message').innerHTML = msg;

    });


    // helper

    function addQueryStringToUrl(url, ...args) {
      var tmp = url + '?';
      if (args.length < 3) {
        // console.log(args);
        tmp = `${tmp}${args[0]}=${args[1]}`;
      }
      else {
        tmp = `${tmp}${args[0]}=${args[1]}`;
        for (let i = 2; i < args.length; i += 2) {
          tmp += `&${args[i]}=${args[i + 1]}`;
        }
      }
      return tmp;
    }
